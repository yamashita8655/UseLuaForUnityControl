--Unity側から呼ばれ、Unity側の処理を呼び出す--Unity〜というのは、Unity側から関数の登録がされていないと使えない（というか、処理がない）--LuaのMain関数みたいな奴function LuaMain()	--CsvManagerの初期化を行う	CsvManagerScriptName = "CsvManager"	LoadLuaFile(CsvManagerScriptName)	UnityInitCsvManager(CsvManagerScriptName)	--タイトルシーンの初期化を行う。	InitTitleScene()		--仮で、試しに処理をリアルタイムに切り替えてみる	--InitShopBillingPoint()end--タイトルシーンの作成function InitTitleScene()	--タイトルシーンに移動する	LuaLoadLevel("TitleScene")end--課金ポイント画面の作成function InitShopBillingPoint()	--課金ポイントを表示する為の処理		--まず、その画面のハンドリングを行うLuaScriptの読み込み	LoadLuaFile("ShopBillingPoint")	LuaBindCommonFunction("ShopBillingPoint")	LuaCallLuaFunction("ShopBillingPoint", "Init")end--プレハブのロードfunction LuaLoadPrefab(prefabName)	UnityLoadPrefab(prefabName)end--シーンの切り替えfunction LuaLoadLevel(sceneName)	UnityLoadLevel(sceneName)end--他のLuaの関数を呼び出すfunction LuaCallLuaFunction(fileName, functionName)	UnityCallLuaFunction(fileName, functionName)end--基本Lua関数を使えるようにするfunction LuaBindCommonFunction(fileName)	UnityBindCommonFunction(fileName)end--Luaファイルを読み込んで、アクセスできる状態にするfunction LoadLuaFile(filename)	UnityLoadLuaFile(filename)end
