--単純なパラメータだけ渡して計算させて、計算結果を返す
--相手に与えるダメージを返す
function CalcNormalAttackDamage(myHp, myAttack, myDefense, enemyHp, enemyAttack, enemyDefense)
	--とりあえず、仕様的には攻撃者の攻撃力から防御者の防御力を引いた値がダメージになるとする
	--但し、最低保障ダメージは1にする
	UnityDebugLog(myHp)

	damage = myAttack - enemyDefense
	if damage < 0 then
		damage = 1
	end

	return damage
end

--スペシャル攻撃の計算方法
function CalcSpecialAttackDamage(myHp, myAttack, myDefense, enemyHp, enemyAttack, enemyDefense)
	--このスペシャル攻撃は、自分のHpの10％を消費して、消費したHP量の2倍のダメージを、通常攻撃に加算する
	addDamage = myHp/10
	damage = (myAttack + addDamage*2) - enemyDefense
	if damage < 0 then
		damage = 1
	end

	myDamage = addDamage

	return damage, myDamage
end
