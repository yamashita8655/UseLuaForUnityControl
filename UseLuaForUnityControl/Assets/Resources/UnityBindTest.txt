--Unity側から呼ばれ、Unity側の処理を呼び出す--Unity〜というのは、Unity側から関数の登録がされていないと使えない（というか、処理がない）

--LuaのMain関数みたいな奴
function LuaMain()
	LuaChangeScene("Title", "MainCanvas")
	LuaLoadPrefabAfter("Prefabs/System/FadeObject", "SystemCanvas")
	LuaLoadPrefabAfter("Prefabs/HeaderObject", "HeaderFooterCanvas")
	LuaLoadPrefabAfter("Prefabs/FooterObject", "HeaderFooterCanvas")
	LuaSetActive("Prefabs/HeaderObject", false)
	LuaSetActive("Prefabs/FooterObject", false)
end

--アニメーション再生
function LuaPlayAnimator(objectName, animationName, callbackMethodName)
	UnityPlayAnimator(objectName, animationName, callbackMethodName)
end

--シーンオブジェクトの切り替え
function LuaChangeScene(sceneName, parentName)
	UnityChangeScene(sceneName, parentName)
end

--アクティブの切り替え
function LuaSetActive(prefabName, active)
	UnitySetActive(prefabName, active)
end

--プレハブをロードするだけ
function LuaLoadPrefabAfter(prefabName, parentName)
	UnityLoadPrefabAfter(prefabName, parentName)
end

--プレハブのロード
function LuaLoadPrefab(prefabName)
	UnityLoadPrefab(prefabName)
end

--シーンの切り替え
function LuaLoadLevel(sceneName)
	UnityLoadLevel(sceneName)
end

--他のLuaの関数を呼び出す
function LuaCallLuaFunction(fileName, functionName)
	UnityCallLuaFunction(fileName, functionName)
end

--基本Lua関数を使えるようにする
function LuaBindCommonFunction(fileName)
	UnityBindCommonFunction(fileName)
end

--Luaファイルを読み込んで、アクセスできる状態にする
function LoadLuaFile(filename)
	UnityLoadLuaFile(filename)
end

--Unity側から呼び出される。Event系の関数
function EventClickButton(eventName)
	if eventName == "TitleScene_ClickHome" then
		LuaPlayAnimator("Prefabs/System/FadeObject", "FadeIn", "TitleScene_ClickHome_Callback")
	elseif eventName == "Footer_ClickHome" then
		LuaPlayAnimator("Prefabs/System/FadeObject", "FadeIn", "Footer_ClickHome_Callback")
	elseif eventName == "Footer_ClickCustom" then
		LuaPlayAnimator("Prefabs/System/FadeObject", "FadeIn", "Footer_ClickCustom_Callback")
	elseif eventName == "Footer_ClickQuest" then
		LuaPlayAnimator("Prefabs/System/FadeObject", "FadeIn", "Footer_ClickQuest_Callback")
	elseif eventName == "Footer_ClickOption" then
		LuaPlayAnimator("Prefabs/System/FadeObject", "FadeIn", "Footer_ClickOption_Callback")
	end
end

--タイトルシーン関数
--ホームへ行く
function TitleScene_ClickHome_Callback()
	LuaChangeScene("Home", "MainCanvas")
	LuaSetActive("Prefabs/HeaderObject", true)
	LuaSetActive("Prefabs/FooterObject", true)
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end

--フッターボタン関数
function Footer_ClickHome_Callback()
	LuaChangeScene("Home", "MainCanvas")
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end
function Footer_ClickHome_Callback()
	LuaChangeScene("Home", "MainCanvas")
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end
function Footer_ClickCustom_Callback()
	LuaChangeScene("Custom", "MainCanvas")
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end
function Footer_ClickQuest_Callback()
	LuaChangeScene("Quest", "MainCanvas")
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end
function Footer_ClickOption_Callback()
	LuaChangeScene("Option", "MainCanvas")
	LuaPlayAnimator("Prefabs/System/FadeObject", "FadeOut", "")
end
